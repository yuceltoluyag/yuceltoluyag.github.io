#### Python

```python
def merhaba_dunya():
    print("Merhaba Dünya!")

if __name__ == "__main__":
    merhaba_dunya()

# Sınıf örneği
class Kisi:
    def __init__(self, ad, yas):
        self.ad = ad
        self.yas = yas

    def selamla(self):
        return f"Merhaba, ben {self.ad}. {self.yas} yaşındayım."
```
